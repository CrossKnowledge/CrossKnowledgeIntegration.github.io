### Example
```xml
<providers>

        <columns>
            <trainingId/>
            <trainingStatus/>
            <sessionId/>
            <sessionGuid/>
            <modality/>
            <title/>
            <description/>
            <startDate/>
            <endDate/>
            <creationDate/>
            <pathModificationDate/>
            <modificationDate/>
            <registeredLearners/>
            <maxNbLearners/>
            <constantValue/>
            <hasClassroomActivity/>

            <dateFormat>MM-DD-YYYY</dateFormat>
        </parameters>

</providers>
```
